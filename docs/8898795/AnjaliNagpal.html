x = [1,2,3,4,5]
y = [3,3,3,3,3]

plt.plot(x, y, label = "Horizontal line", linestyle="-")
plt.plot(y, x, label = "Vertical line", linestyle="--")
plt.plot(x, np.sin(x), label = "Sin(x)", linestyle="-.")
plt.plot(x, np.cos(x), label = "Cos(x)", linestyle=":")
plt.legend()
plt.show()
