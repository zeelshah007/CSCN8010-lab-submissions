Bargraph using Matplotlib¶

This bargraphs illustrate the price percentage increase on electronic Items in Canada from 2022 to 2023¶

In [5]:
import matplotlib.pyplot as plt
import numpy as np

fig, ax = plt.subplots(figsize=(5, 6))
electronicItems = ['Mobile Phones', 'Washing Machine', 'Television', 'Gaming Consoles']
price = [30,45,70,60]

ax.bar(electronicItems, price, width = 0.5,color = 'green')
plt.xlabel('Electronic Items')
plt.ylabel('Percentage of Increase')
plt.show()
